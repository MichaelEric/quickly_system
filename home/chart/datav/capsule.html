# 胶囊柱图

<dv-capsule-chart :config="config" style="width:300px;height:200px" />

# 基本示例

<template>
  <div class="chart-container">
    <dv-capsule-chart :config="config" style="width:300px;height:200px" />
  </div>
</template>

<script>
export default {
  data() {
    return {
      config: {
        data: [
          {
            name: "天津",
            value: 456,
          },
          {
            name: "北京",
            value: 123,
          },
          {
            name: "广州",
            value: 98,
          },
          {
            name: "上海",
            value: 75,
          },
          {
            name: "深圳",
            value: 66,
          },
        ],
        colors: ["#e062ae", "#fb7293", "#e690d1", "#32c5e9", "#96bfff"],
        // unit: "单位",
        showValue: true,
      },
    };
  },
};
</script>

# config 属性

属性 说明 类型 可选值 默认值
data 柱数据 Array<object> data属性 []
unit 单位 String '55%' -
colors 环颜色 Array<String> hex|rgb|rgba ['#37a2da', '#32c5e9', '#67e0e3', '#9fe6b8', '#ffdb5c', '#ff9f7f', '#fb7293']
showValue 显示数值 Boolean - false

# data 属性

属性 说明 类型 可选值 默认值
name 柱名称 String - -
value 柱对应值 Number - -
